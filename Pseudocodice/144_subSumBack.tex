\caption{\protect\Boolean \protect\SsRec{\protect[] A, \protect\Int i, \protect\Int r}}
\uIf{r == 0}{
	\Return \True\;
}
\uElseIf{i == 0}{
	\Return \False\;
}
\uElseIf{A[i] $>$ r}{
	\Return \SsRec{A, i - 1, r}\;
}
\Else{
	\Return \SsRec{A, i - 1, r} \Or \SsRec{A, i - 1, r - A[i]}\;
}
