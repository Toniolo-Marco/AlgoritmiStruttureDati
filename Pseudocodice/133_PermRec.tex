\uIf{missing == 0}{
	\ProcessSolution{S, i - 1}\;
}
\ElseIf{i $\le$ n \And 0 $<$ missing $\le$ n - (i - 1)}{
	\ForEach{c $\in$ $\{0, 1\}$}{
		S[i] = c\;
		\PermRec{n, missing - c, S, i + 1}\;
	}
