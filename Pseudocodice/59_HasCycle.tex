\caption{\protect\HasCycle{\protect\Graph G, \protect\Node u, \protect\Int$\&$ time, \protect\Int[] dt, \protect\Int[] ft}}

time += 1\;
dt[u] = time\;
\ForEach{v $\in$ G.\Adj{u}}{
	\uIf{dt[v] == 0}{
		\If{\HasCycle{G, v, time, dt, ft}}{
			\Return \True\;		
		}
	}
	\ElseIf{dt[u] $>$ dt[v] \And ft[v] == 0}{
		\Return \True\;	
	}
}
time +=1\;
ft[u] = time\;
\Return \False\;
