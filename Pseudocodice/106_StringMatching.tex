\caption{\protect\Int \protect\StringMatching{\protect\Item[] P, \protect\Item[] T, \protect\Int m, \protect\Int n}}
\Int[][] DP = \New \Int$[0\dots m][0\dots n]$\;
\For{\Int j = 0 \To n}{
	DP[0][j] = 0\;
}
\For{\Int i = 1 \To m}{
	DP[i][0] = i\;
}
\For{\Int i = 1 \To m}{
	\For{\Int j = 1 \To n}{
		DP[i][j] = \Min{DP[i-1][j-1] + iif(P[i] == T[j], 0, 1), DP[i-1][j]+1, DP[i][j-1]+1}\;
	}
}
\Int pos = 0\;
\For{\Int j = 1 \To n}{
	\If{DP[m][j] $<$ DP[m][pos]}{
		pos = j\;
	}
}
\Return pos\;
