\caption{\protect\MaxHeapRestore{\protect\Item[] A, \protect\Int i, \protect\Int i}}

\Int max = i\;
\If{\L{i} $\le$ dim \And A[\L{i}] $>$ A[max]}{
	max = \L{i}\;
}
\If{\R{i} $\le$ dim \And A[\R{i}] $>$ A[max]}{
	max = \R{i}\;
}
\If{i $\neq$ max}{
	A[i] $\leftrightarrow$ A[max]
	\MaxHeapRestore{A, max, dim}\;
}
