\caption{\protect\Int[] \protect\ComputePredecessor{\protect\Int[] a, \protect\Int[] b, \protect\Int n}}
\Int[] pred = \New \Int$[0\dots n]$\;
pred[0] = 0\;
\For{\Int i = 1 \To n}{
	\Int j = i - 1\;
	\While{j $>$ 0 \And b[j] $>$ a[i]}{
		j = j - 1\;
	}
	pred[i] = j\;
}
\Return pred\;
