\caption{\protect\Decrease{\protect\PriorityItem x, \protect\Int p}}

\Precondition p $<$ x.priority\;
x.priority = p\;
\Int i = x.pos\;
\While{i $>$ 1 \And H[i].priority $<$ H[\P{i}].priority}{
	\Swap{H, i, \P{i}}\;
	i = \P{i}\;
