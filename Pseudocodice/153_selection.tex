\caption{\protect\Item \protect\Selection{\protect\Item[] A, \protect\Int start, \protect\Int end, \protect\Int k}}
\uIf{start == end}{
	\Return A[start]\;
}
\Else{
	\Int j = \Pivot{A, start, end}\;
	\Int q = j - start + 1\;
	\uIf{k == q}{
		\Return A[j]\;
	}
	\uElseIf{k $<$ q}{
		\Return \Selection{A, start, j - 1, k}\;
	}
	\Else{
		\Return \Selection{A, j + 1, end, k - q}\;
	}
}
