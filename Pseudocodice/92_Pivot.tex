\caption{\protect\Int \protect\Pivot{\protect\Item[] A, \protect\Int start, \protect\Int end}}
\Item p = A[start]\;
\Int j = start\;
\For{\Int i = start +1 \To end}{
	\If{A[i] $<$ p}{
		j += 1\;
		A[i] $\leftrightarrow$ A[j]\;	
	}
}
A[start] = A[j]\;
A[j] = p\;
\Return j\;
