\caption{(\protect\Int[], \protect\Int[])  \protect\ProCamMin{\protect\Graph G}}
\Int[] d = \New \Int$[1\dots G.n]$\;
\Int[] T = \New \Int$[1\dots G.n]$\;
\ForEach{u, v $\in$ G.V() - $\{s\}$}{
	T[u][v] = \Nil\;
	d[u][v] = $+\infty$\;
}
T[s] = \Nil\;
d[s] = 0\;
\ForEach{u $\in$ G.V()}{
	\ForEach{v $\in$ G.\Adj{u}}{
		d[u][v] = G.w(u, v)\;
		T[u][v] = u\;
	}
}
\For{\Int k = 1 \To G.n}{
	\ForEach{u $\in$ G.V()}{
		\ForEach{v $\in$ G.V()}{
			\If{d[u][k] + d[k][v] $<$ d[u][v]}{
				d[u][v] = d[u][k] + d[k][u]\;
				T[u][v] = T[k][v]\;
			}
		}
	}
}
\Return (T, d)\;
