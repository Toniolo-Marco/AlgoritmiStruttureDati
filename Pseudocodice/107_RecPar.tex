\caption{\protect\Int \protect\RecPar{\protect\Item[] c, \protect\Int i, \protect\Int j}}
\uIf{i == j}{
	\Return 0\;
}
\Else{
	\Int min = $+\infty$\;
	\For{\Int k = i \To j - 1}{
		\Int q = \RecPar(c, i, k) + \RecPar(c, k + 1, j) +c[i-1]$\cdot$c[k]$\cdot$c[j]\;
		\If{q $<$ min}{
			min = q\;
		}
	}
	\Return min\;
}
