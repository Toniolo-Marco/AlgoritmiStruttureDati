	\If{p[i].y $<$ p[min].y}{
		min = i\;
	}
}

$\{$riordina p$[2,\dots n]$ in base all'angolo formato rispetto all'asse orizzontale quando sono connessi con p[1]$\}$\;
$\{$elimina eventuali punti ``allineati" tranne i pi\`u lontani da $p_1$ aggioirnando $n\}$
\Stack S = \StackCos{}\;
S.\Push{$p_1$}\;

	\While{\Not \SameSide{S.\Top{}, S.\Tops{}, $p_1$, $p_i$}}{
