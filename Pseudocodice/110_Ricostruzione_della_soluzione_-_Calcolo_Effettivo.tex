\caption{Ricostruzione della soluzione - Calcolo Effettivo}
\Int[][]\Space\Fn{\Multiply{\Matrix[] A, \Int[][] S, \Int i, \Int j}}{
	\uIf{i == j}{
		\Return A[i]\;
	}
	\Else{
		\Int[][] X = \Multiply{S, i, S[i][j]}\;
		\Int[][] Y = \Multiply{S, S[i][j]+1, j}\;
		\Return \MMultiplication{X, Y}\;
	}
