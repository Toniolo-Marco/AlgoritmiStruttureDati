\caption{\protect\Stack \protect\TopSort{\protect\Graph G}}

\Stack S = \StackCos{}\;
\Boolean[] visited = \Boolean[G.\Size{}]\;
\ForEach{u $\in$ G.\V{}}{
	visited[u] = \False\;
}
\ForEach{u $\in$ G.\V{}}{
	\If{\Not visited[u]}{
		\TsDfs{G, visited, S}\;
	}
}
\Return S\;
