\caption{(\protect\Int[], \protect\Int[])  \protect\ProCamMin{\protect\Graph G, \protect\Node s}}
\PriorityQueue Q = \PriorityQueueCos{}\;
Q.\Insert{s, 0}\;
\While{\Not Q.\IsEmpty{}}{
	\Int u = Q.\DeleteMin{}\;
	b[u] = \False\;
	\ForEach{v $\in$ G.\Adj{u}}{
		\If{d[u]+G.w(u, v) $<$ d[v]}{
			\uIf{\Not b[v]}{
				Q.\Insert{v, d[u]+G.w(u, v)}\;
				b[v] = \True\;
			}
			\Else{
				Q.\Decrease{v, D[u]+G.w(u, v)}\;
			}
			T[v] = u\;
			d[v] = d[u] + G.w(u, v)\;
		}
	}
}
