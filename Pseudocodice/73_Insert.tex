\caption{\protect\PriorityItem \protect\Insert{\protect\Item x, \protect\Int p}}

\Precondition dim $<$ capacity\;
dim += 1\;
H[dim] = \New \PriorityItemCos{}\; 
H[dim].value = x\;
H[dim].priority = p\;
H[dim].pos = dim\;
\Int i = dim\;
\While{i $>$ 1 \And H[i].priority $<$ H[\P{i}].priority}{
	\Swap{H, i, \P{i}}\;
	i = \P{i}\;
}
\Return H[i]\;
