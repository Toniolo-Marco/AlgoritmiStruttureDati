\caption{\protect\Int \protect\LCS{\protect\Item[] T, \protect\Item[] U, \protect\Int n, \protect\Int m}}
\Int[][] DP = \New \Int$[0\dots n][0\dots m]$\;
\For{\Int i = 0 \To n}{
	DP[0][n] = DP[n][0] = 0\;
}
\For{\Int i = 1 \To n}{
	\For{\Int j = 1 \To m}{
		\uIf{T[i] == U[j]}{
			DP[i][j] = DP[i - 1][j - 1] +1\;		
		}
		\Else{
			DP[i][j] = \Max{DP[i - 1][j], DP[i][j - 1]}\;		
		}
	}
}
\Return DP[n][m]\;
