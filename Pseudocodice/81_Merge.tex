\caption{\protect\Merge{\protect\Int x, \protect\Int y}}
\Item $r_x$ = \Find{x}\;
\Item $r_y$ = \Find{y}\;
\If{$r_x\ \neq\ r_y$}{
	\uIf{rank[$r_x$] $>$ rank[$r_y$]}{
		parent[$r_y$] = $r_x$\;	
	}
	\uIf{rank[$r_y$] $>$ rank[$r_x$]}{
		parent[$r_x$] = $r_y$\;	
	}
	\Else{
			parent[$r_x$] = $r_y$\;
			rank[$r_y$] += 1\;
	}
}
