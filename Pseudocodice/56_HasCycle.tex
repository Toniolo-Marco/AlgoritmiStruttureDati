\caption{\protect\Boolean[] \protect\HasCycle{\protect\Graph G, \protect\Node u, \protect\Node p, \protect\Boolean[] visited}}

visited[u] = \True\;
\ForEach{v $\in$ G.\Adj{u} - $\{$p$\}$}{
	\uIf{visited[u]}{
		\Return \True\;	
	}
	\ElseIf{\HasCycle{G, v, u, visited}}{
		\Return \True\;	
	}
}
\Return \False\;
