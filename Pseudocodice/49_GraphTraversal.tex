\caption{\protect\GraphTraversal{\protect\Graph G, \protect\Node r}}

\Set Q = \SetCos{} \;
Q.\Insert{r}\;
\Boolean[] visited = \New \Boolean[G.\Size{}]\;
\ForEach{u $\in$ G.\V{} - $\{$r$\}$}{
	visited[u] = \False\;
}
visited[r] = \True\;
\While{\Not Q.\Empty{}}{
	\Node u = Q.\Remove{}\;
	$\{$Visita il nodo u$\}$\;
	\ForEach{v $\in$ G.\Adj{u}}{
		$\{$Visita l'arco (u,v)$\}$\;
		\If{\Not visited[v]}{
			visited[v] = \True\;
			Q.\Insert{v}\;
		}
	}
}
\end{algorithm}
