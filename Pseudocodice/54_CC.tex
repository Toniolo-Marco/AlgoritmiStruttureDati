\caption{\protect\Int[] \protect\CC{\protect\Graph G}}

\Int[] id = \New \Int[G.\Size{}]\;
\ForEach{u $\in$ G.\V{}}{
	id[u] = 0\;
}
\Int counter = 0\;
\ForEach{u $\in$ G.\V{}}{
	\If{id[u] == 0}{
		counter += 1\;
		\CCdfs{G, counter, u, id}\;	
	}
}
\Return id\;
