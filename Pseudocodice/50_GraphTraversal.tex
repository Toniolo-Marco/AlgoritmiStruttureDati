\caption{\protect\GraphTraversal{\protect\Graph G, \protect\Node r, \protect\Int[] erdos, \protect\Node[] parents}}

\Set Q = \SetCos{}\; 
Q.\Insert{r}\;
\ForEach{u $\in$ G.\V{} - $\{$r$\}$}{
	erdos[u] = $\infty$\;
}
erdos[r] = 0\;
parents[r] = \Nil\;
\While{\Not Q.\Empty{}}{
	\Node u = Q.\Remove{}\;
	$\{$Visita il nodo u$\}$\;
	\ForEach{v $\in$ G.\Adj{u}}{
		\If{erdos[v] = $\infty$}{
			erdos[v] = erdos[u] + 1\;
			parents[v] = u\;			
			Q.\Insert{v}\;
		}
	}
}
\end{algorithm}
