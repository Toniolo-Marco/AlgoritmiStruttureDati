\caption{\protect\Int[] \protect\Transpose{\protect\Graph G}}

\Graph $G^T$ = \GraphCos{G}\;
\ForEach{u $\in$ G.\V{}}{
	$G^T$.\InsertNode{u}\;
}
\ForEach{u $\in$ G.\V{}}{
	\ForEach{v $\in$ G.\Adj{u}}{
		$G^T$.\InsertEdge{v, u}
	}
}
\Return $G^T$\;
