\caption{\protect\Boolean \protect\Sudoku{\protect\Int[][] S, \protect\Int i}}
\If{i == 81}{
	\ProcessSolution{S, n}\;
	\Return \True\;
}
\Int x = i $\mod 9$\;
\Int y = $\lfloor\frac{i}{9}\rfloor$\;
\Set C = \Choices{S, x, y}\;
\Int old = S[x][y]\;
\ForEach{c $\in$ C}{
	S[x][y] = c\;
	\If{\Sudoku{S, i + 1}}{
		\Return \True\;
	}
}
S[x][y] = old\;
\Return \False\;
