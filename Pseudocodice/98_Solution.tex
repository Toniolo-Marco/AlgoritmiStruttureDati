\caption{\protect\Set \protect\Solution{\protect\Int[] DP, \protect\Int[] D, \protect\Int i}}
\uIf{ i == 0}{
	\Return $\emptyset$\;
}
\uElseIf{i == 1}{
	\Return $\{1\}$\;
}
\uElseIf{DP[i] == DP[i-1]}{
	\Return \Solution{DP, D, i - 1}\;
}
\Else{
	\Set sol = \Solution(DP, D, i - 2)\;
	sol.\Insert{i}\;
	\Return sol\;
