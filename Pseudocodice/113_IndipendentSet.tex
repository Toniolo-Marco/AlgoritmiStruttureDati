\caption{\protect\Set \protect\IndipendentSet{\protect\Int[] a, \protect\Int[] b}}
$\{$ordina $a$ e $b$ in modo che $b[1]\le b[2]\le\cdots\le b[n]$ $\}$\;
\Set S = \SetCos{}\;
S.\Insert{1}\;
\Int last = 1\;
\For{\Int i = 2 \To n}{
	\If{a[i] $\ge$ b[last]}{
		S.\Insert{i}\;
		last = i\;
	}
}
