\caption{\protect\Boolean[] \protect\HasCycle{\protect\Graph G}}

\Boolean[] visited = \New \Boolean[G.\Size{}]\;
\ForEach{u $\in$ G.\V{}}{
	visited[u] = \False\;
}
\ForEach{u $\in$ G.\V{}}{
	\If{\Not visited[u]}{
		\If{\HasCycleRec{G, u, \Nil, visited}}{
			\Return \True\;	
		}
	}
}
\Return \False\;
