\caption{\protect\Boolean \protect\Cavallo{\protect\Int[][] S, \protect\Int i, \protect\Int x, \protect\Int y}}
\comment{Se $i=64$ ho fatto $63$ mosse e ho completato un tour aperto}
\If{i == 64}{
	\ProcessSolution{S}\;
	\Return \True\;
}
\Set C = \Choices{S, x, y}\;
\ForEach{c $\in$ C}{
	S[x][y] = i\;
	\If{\Cavallo{S, i + 1, x + $m_x$[c], y + $m_y$[c]}}{
		\Return \True\;
	}
	S[x][y] = 0\;
}
