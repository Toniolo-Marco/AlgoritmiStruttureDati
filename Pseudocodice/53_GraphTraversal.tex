\caption{\protect\GraphTraversal{\protect\Graph G, \protect\Node r}}

\Set S = \SetCos{} \;
S.\Insert{r}\;
\Boolean[] visited = \New \Boolean[G.\Size{}]\;
\ForEach{u $\in$ G.\V{}}{
	visited[u] = \False\;
}

\While{\Not S.\Empty{}}{
	\Node u = S.\Remove{}\;
	\If{\Not visited[u]}{
		$\{$Visita il nodo u in pre-order$\}$\;
		visited[u] = \True\;
		\ForEach{v $\in$ G.\Adj{u}}{
			$\{$Visita l'arco (u,v)$\}$\;
			S.\Insert{v}\;	
	}	
	}
}
\end{algorithm}
