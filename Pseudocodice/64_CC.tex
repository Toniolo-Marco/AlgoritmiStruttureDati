\caption{\protect\Int[] \protect\CC{\protect\Graph G, \protect\Stack S}}

\Int[] id = \New \Int[G.\Size{}]\;
\ForEach{u $\in$ G.\V{}}{
	id[u] = 0\;
}
\Int counter = 0\;
\While{\Not S.\Empty{}}{
	u = S.\Pop{}\;
	\If{id[u] == 0}{
		counter += 1\;
		\CCdfs{G, counter, u, id}\;
	}
}
\Return id\;
