	S[i] = c\;
	\Int lb = \Lb{$<$dati problema$>$, S, i, $<$dati parziali$>$}\;
	\If{lb $<$ minCost}{
		\uIf{i $<$ n}{
			\BranchBound{$<$dati problema$>$, S, i + 1, $<$dati parziali$>$}\;
		}
		\Else{
			\If{\Cost{S, i} $<$ minCost}{
				minSol = S\comment{Variabile globale}
